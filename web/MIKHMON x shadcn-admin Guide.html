Setup
Project Overview
shadcn-admin menggunakan Tailwind CSS v4 dengan token via CSS custom properties. Strategi kita: remapping token + tambah komponen baru — tanpa menyentuh komponen Radix UI yang sudah ada.
Non-destructive: Hanya mengubah theme.css dan menambah file baru. Semua komponen Radix tetap berfungsi.
Tailwind v4: Token didefinisikan di dalam @theme inline {} di theme.css — bukan di tailwind.config.js.

function ResourceBar({ label, value }: { label: string; value: number }) {
  const cls = value >= 80 ? 'danger' : value >= 60 ? 'warn' : ''
  return (
    <div className="space-y-1.5">
      <div className="flex justify-between text-sm">
        <span className="font-semibold">{label}</span>
        <span className="font-mono text-muted-foreground">{value}%</span>
      </div>
      <div className="mk-progress">
        <div className={`mk-progress-fill ${cls}`} style={{ width: `${value}%` }} />
      </div>
    </div>
  )

// src/hooks/use-ping.ts
import { useState, useEffect } from 'react'
export type PingStatus = 'excellent' | 'good' | 'high' | 'timeout'
export interface PingResult { ms: number | null; status: PingStatus }

function getStatus(ms: number | null): PingStatus {
  if (ms === null) return 'timeout'
  if (ms < 30)  return 'excellent'
  if (ms < 80)  return 'good'
  return 'high'
}

export function usePing(intervalMs = 2000) {
  const [result, setResult] = useState<PingResult>({ ms: null, status: 'timeout' })
  const [base, setBase] = useState(18)
  useEffect(() => {
    let cancelled = false
    const tick = async () => {
      // TODO: replace with real API call
      // const { ms } = await fetch('/api/ping').then(r => r.json())
      await new Promise(r => setTimeout(r, 200))
      if (cancelled) return
      if (Math.random() < 0.04) { setResult({ ms: null, status: 'timeout' }); return }
      let ms = base + (Math.random() - 0.4) * 12
      if (Math.random() < 0.08) ms += 60 + Math.random() * 80
      ms = Math.max(1, Math.round(ms))
      setResult({ ms, status: getStatus(ms) })
      setBase(b => Math.max(5, Math.min(40, b + (Math.random() - 0.5) * 3)))
    }
    tick()
    const id = setInterval(tick, intervalMs)
    return () => { cancelled = true; clearInterval(id) }
  }, [intervalMs])
  return result
}
